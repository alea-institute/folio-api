  <action>
Replace the Plan 03 stub `function expand(iri) { return Promise.reject(...); }` with:

    function expand(iri) {
        if (!state.graphData) return Promise.reject(new Error('No graph loaded'));
        if (state.expandedIris.has(iri)) return Promise.resolve(state.graphData);  // dedupe

        var url = '/explore/api/entity-graph/' + encodeURIComponent(iri) + '?mode=children';
        return fetch(url, { headers: { 'Accept': 'application/json' } })
            .then(function (res) {
                if (!res.ok) throw new Error('HTTP ' + res.status);
                return res.json();
            })
            .then(function (data) {
                // Mark this node as expanded; merge children into a per-node bucket.
                state.expandedIris.add(iri);
                state.graphData.expandedChildren = state.graphData.expandedChildren || {};
                state.graphData.expandedChildren[iri] = data.children;
                // The selected node's children also flow into the legacy `children` field
                // so buildELKGraph (Plan 07) can keep its current edge-construction logic.
                if (iri === state.graphData.selected.iri) {
                    state.graphData.children = data.children;
                }
                // Rebuild + re-layout. buildELKGraph must be extended (next bullet) to also
                // include edges from any IRI in expandedIris to its children.
                var spec = buildELKGraph(state.graphData);
                return runLayout(spec).then(function (laid) {
                    state.graphData.layout = laid;
                    // Re-render WITHOUT calling fitGraph — preserve current transform (D-26).
                    _mountGraph(state.graphData, { preserveZoom: true });
                    return state.graphData;
                });
            })
            .catch(function (err) {
                if (window.console && window.console.error) {
                    window.console.error('[EntityGraph] expand failed for', iri, err);
                }
                throw err;
            });
    }

Update buildELKGraph (Plan 07) to also include children from state.graphData.expandedChildren:

    // After the existing children-of-selected loop, iterate expandedChildren:
    if (graphData.expandedChildren) {
        Object.keys(graphData.expandedChildren).forEach(function (parentIri) {
            // Skip if parent is the selected node — already covered by the selected→children loop.
            if (parentIri === graphData.selected.iri) return;
            (graphData.expandedChildren[parentIri] || []).forEach(function (c, i) {
                edges.push({ id: 'e_exp_' + parentIri + '_' + i, sources: [parentIri], targets: [c.iri] });
                targetIds.add(c.iri);
                allNodes.push(c);   // include in the children loop so an ELK node entry is created
            });
        });
    }

Update _mountGraph to honor `{ preserveZoom: true }`: skip the fitGraph call but still call applyTransform (so the new DOM picks up the existing transform).

Behavior summary:
- First click on +N Children (selected node): merges children into graphData.children; re-layout + re-render keeps current zoom.
- Subsequent click on a child's hover badge: merges into expandedChildren[parentIri]; same re-layout + re-render.
- Refresh (refreshFor with new entity): expandedIris cleared (Plan 07's reset block); previously expanded children DISCARDED (D-11).
